package com.drpicox.game.command.decorators;

import com.drpicox.game.rooms.Direction;
import com.fasterxml.jackson.annotation.JsonValue;

import java.util.LinkedHashMap;
import java.util.Map;

public class PlayerHint {

    private Direction direction;
    private String type;

    public PlayerHint(Direction direction, String type) {
        this.direction = direction;
        this.type = type;
    }

    @JsonValue
    public Map getJsonObject() {
        Map result = new LinkedHashMap();
        result.put("direction", direction.name().toLowerCase());
        result.put("type", type);
        return result;
    }
}
