package com.drpicox.game.command.decorators;

import com.drpicox.game.command.CommandRequest;
import com.drpicox.game.command.CommandResponse;
import com.drpicox.game.players.Player;

import java.util.ArrayList;
import java.util.Arrays;
import java.util.Collections;
import java.util.List;

public class DecoratorsList implements CommandResponseDecorator {

    private List<CommandResponseDecorator> list = new ArrayList<>();

    public DecoratorsList(CommandResponseDecorator ...decorators) {
        this.list.addAll(Arrays.asList(decorators));
    }

    @Override
    public void decorate(CommandResponse response, Player player, CommandRequest request) {
        list.forEach(decorator -> decorator.decorate(response, player, request));
    }
}
